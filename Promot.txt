BACKEND API DETAILS (REST)
--------------------------------------------------

AUTH ENDPOINTS

POST /api/auth/register
- Description: Register new user (Student/Tutor)
- Auth: None
- Request Body: { name: string, email: string, password: string, role: "student"|"tutor", grade?: number, subjects?: string[] }
- Response: { success: true, user: User, token: string, refreshToken: string }
- Errors: 400 (validation), 409 (email exists)

POST /api/auth/login
- Description: User login
- Auth: None
- Request Body: { email: string, password: string }
- Response: { success: true, user: User, token: string, refreshToken: string }
- Errors: 401 (invalid credentials), 403 (account disabled)

POST /api/auth/refresh
- Description: Refresh access token
- Auth: None (requires refresh token in body)
- Request Body: { refreshToken: string }
- Response: { success: true, token: string }
- Errors: 401 (invalid refresh token)

POST /api/auth/logout
- Description: Logout user
- Auth: Bearer token
- Request Body: { refreshToken: string }
- Response: { success: true, message: string }
- Errors: 401 (invalid token)

USER ENDPOINTS

GET /api/users/me
- Description: Get current user profile
- Auth: Bearer token
- Query Params: None
- Response: { success: true, user: User }
- Errors: 401 (unauthorized)

GET /api/users/:id
- Description: Get user by ID (public profile for students/tutors)
- Auth: Bearer token
- Path Params: id (user ID)
- Response: { success: true, user: PublicUser }
- Errors: 401 (unauthorized), 404 (user not found)

PUT /api/users/:id
- Description: Update user profile (own profile only)
- Auth: Bearer token
- Path Params: id (user ID)
- Request Body: { name?: string, bio?: string, avatar?: string, subjects?: string[] }
- Response: { success: true, user: User }
- Errors: 401 (unauthorized), 403 (not own profile), 404 (user not found)

GET /api/admin/users
- Description: Get all users (Admin only)
- Auth: Bearer token (Admin)
- Query Params: page, limit, role, status, search
- Response: { success: true, users: User[], total: number, page: number }
- Errors: 401 (unauthorized), 403 (not admin)

PUT /api/admin/users/:id/status
- Description: Update user status (Admin only)
- Auth: Bearer token (Admin)
- Path Params: id (user ID)
- Request Body: { status: "active"|"inactive"|"suspended", reason?: string }
- Response: { success: true, user: User }
- Errors: 401 (unauthorized), 403 (not admin), 404 (user not found)

COURSE ENDPOINTS

POST /api/courses
- Description: Create new course (Tutor/Admin)
- Auth: Bearer token (Tutor/Admin)
- Request Body: { title: string, description: string, grade: number, subject: string, price: number, thumbnail?: string, syllabus: string[] }
- Response: { success: true, course: Course }
- Errors: 401 (unauthorized), 403 (not tutor/admin), 400 (validation)

GET /api/courses
- Description: Get courses with filters
- Auth: Bearer token
- Query Params: page, limit, grade, subject, tutor, search, sortBy, sortOrder
- Response: { success: true, courses: Course[], total: number, page: number }
- Errors: 401 (unauthorized)

GET /api/courses/:id
- Description: Get course details
- Auth: Bearer token
- Path Params: id (course ID)
- Response: { success: true, course: Course, materials: Material[], sessions: Session[] }
- Errors: 401 (unauthorized), 404 (course not found)

PUT /api/courses/:id
- Description: Update course (Tutor/Admin)
- Auth: Bearer token (Tutor/Admin)
- Path Params: id (course ID)
- Request Body: { title?: string, description?: string, price?: number, thumbnail?: string, syllabus?: string[] }
- Response: { success: true, course: Course }
- Errors: 401 (unauthorized), 403 (not owner/admin), 404 (course not found)

DELETE /api/courses/:id
- Description: Delete course (Tutor/Admin)
- Auth: Bearer token (Tutor/Admin)
- Path Params: id (course ID)
- Response: { success: true, message: string }
- Errors: 401 (unauthorized), 403 (not owner/admin), 404 (course not found)

SESSION ENDPOINTS

POST /api/sessions
- Description: Create new session (Tutor)
- Auth: Bearer token (Tutor)
- Request Body: { courseId: string, title: string, description: string, scheduledAt: Date, duration: number, maxStudents: number, isPaid: boolean }
- Response: { success: true, session: Session }
- Errors: 401 (unauthorized), 403 (not tutor), 400 (validation), 404 (course not found)

GET /api/sessions
- Description: Get sessions with filters
- Auth: Bearer token
- Query Params: page, limit, courseId, tutorId, studentId, status, upcoming, search
- Response: { success: true, sessions: Session[], total: number, page: number }
- Errors: 401 (unauthorized)

GET /api/sessions/:id
- Description: Get session details
- Auth: Bearer token
- Path Params: id (session ID)
- Response: { success: true, session: Session, attendees: User[], materials: Material[] }
- Errors: 401 (unauthorized), 404 (session not found)

PUT /api/sessions/:id
- Description: Update session (Tutor)
- Auth: Bearer token (Tutor)
- Path Params: id (session ID)
- Request Body: { title?: string, description?: string, scheduledAt?: Date, duration?: number, maxStudents?: number }
- Response: { success: true, session: Session }
- Errors: 401 (unauthorized), 403 (not tutor), 404 (session not found)

DELETE /api/sessions/:id
- Description: Delete session (Tutor)
- Auth: Bearer token (Tutor)
- Path Params: id (session ID)
- Response: { success: true, message: string }
- Errors: 401 (unauthorized), 403 (not tutor), 404 (session not found)

MATERIAL ENDPOINTS

POST /api/materials
- Description: Upload material (Tutor)
- Auth: Bearer token (Tutor)
- Request Body: { courseId: string, title: string, description: string, type: "pdf"|"video"|"ppt"|"document", fileUrl: string, isFree: boolean }
- Response: { success: true, material: Material }
- Errors: 401 (unauthorized), 403 (not tutor), 400 (validation), 404 (course not found)

GET /api/materials/:courseId
- Description: Get materials for course
- Auth: Bearer token
- Path Params: courseId
- Query Params: type, isFree
- Response: { success: true, materials: Material[] }
- Errors: 401 (unauthorized), 404 (course not found)

EVALUATION ENDPOINTS

POST /api/evaluations
- Description: Create evaluation (Tutor)
- Auth: Bearer token (Tutor)
- Request Body: { sessionId: string, studentId: string, grade: number, feedback: string, achievements: string[] }
- Response: { success: true, evaluation: Evaluation }
- Errors: 401 (unauthorized), 403 (not tutor), 400 (validation), 404 (session/student not found)

GET /api/evaluations/student/:id
- Description: Get evaluations for student
- Auth: Bearer token (Student/Tutor/Admin)
- Path Params: id (student ID)
- Query Params: sessionId, limit
- Response: { success: true, evaluations: Evaluation[] }
- Errors: 401 (unauthorized), 403 (not authorized), 404 (student not found)

GET /api/evaluations/session/:id
- Description: Get evaluations for session (Tutor)
- Auth: Bearer token (Tutor)
- Path Params: id (session ID)
- Response: { success: true, evaluations: Evaluation[] }
- Errors: 401 (unauthorized), 403 (not tutor), 404 (session not found)

SUBSCRIPTION ENDPOINTS

POST /api/subscriptions/checkout
- Description: Create subscription checkout (Student)
- Auth: Bearer token (Student)
- Request Body: { planId: string, paymentMethodId: string }
- Response: { success: true, subscription: Subscription, clientSecret: string }
- Errors: 401 (unauthorized), 403 (not student), 400 (validation), 402 (payment failed)

GET /api/subscriptions/status
- Description: Get subscription status
- Auth: Bearer token
- Response: { success: true, subscription: Subscription, plan: Plan }
- Errors: 401 (unauthorized)

POST /api/subscriptions/cancel
- Description: Cancel subscription (Student)
- Auth: Bearer token (Student)
- Request Body: { reason?: string }
- Response: { success: true, subscription: Subscription }
- Errors: 401 (unauthorized), 403 (not student), 400 (no active subscription)

LIVE CLASS ENDPOINTS

POST /api/live/token
- Description: Generate live class token
- Auth: Bearer token
- Request Body: { sessionId: string }
- Response: { success: true, token: string, roomId: string }
- Errors: 401 (unauthorized), 403 (not enrolled), 404 (session not found)

POST /api/live/start
- Description: Start live session (Tutor)
- Auth: Bearer token (Tutor)
- Request Body: { sessionId: string }
- Response: { success: true, session: Session }
- Errors: 401 (unauthorized), 403 (not tutor), 404 (session not found)

POST /api/live/end
- Description: End live session (Tutor)
- Auth: Bearer token (Tutor)
- Request Body: { sessionId: string }
- Response: { success: true, session: Session, recordingUrl?: string }
- Errors: 401 (unauthorized), 403 (not tutor), 404 (session not found)

REPORT ENDPOINTS

GET /api/reports/student/:id
- Description: Get student progress report
- Auth: Bearer token (Student/Tutor/Admin)
- Path Params: id (student ID)
- Query Params: startDate, endDate
- Response: { success: true, report: { attendance: number, grades: Grade[], progress: Progress[] } }
- Errors: 401 (unauthorized), 403 (not authorized), 404 (student not found)

GET /api/reports/tutor/:id
- Description: Get tutor performance report (Tutor)
- Auth: Bearer token (Tutor)
- Path Params: id (tutor ID)
- Query Params: startDate, endDate
- Response: { success: true, report: { sessions: number, students: number, rating: number, earnings: number } }
- Errors: 401 (unauthorized), 403 (not tutor), 404 (tutor not found)

GET /api/admin/analytics
- Description: Get platform analytics (Admin)
- Auth: Bearer token (Admin)
- Query Params: startDate, endDate
- Response: { success: true, analytics: { users: number, sessions: number, revenue: number, growth: GrowthStats[] } }
- Errors: 401 (unauthorized), 403 (not admin)
